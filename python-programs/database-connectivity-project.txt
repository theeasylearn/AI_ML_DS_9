Lecturer's Payment Management System
-----------------------------------

Tables
    course (id, title, fees, duration, description)
    batch (id, courseid, startdate, enddate, classtime)
    subject (id, courseid, title, rate)
    teacher (id, name, mobile, email, gender, qualification, experience)
    lecture (id, teacherid, subjectid, batchid, duration, amount, lecturedate, paymentid = 0)
    payment (id, teacherid, paymentdate, amount)

    teacher management

    reports
